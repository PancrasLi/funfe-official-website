变量、作用域与内存 ​

一、变量声明与赋值 ​

1. 变量声明的方式 ​

javascript
// var声明（不推荐）
var oldWay = 'old';     // 函数作用域，有变量提升
var oldWay = 'repeat';  // 可以重复声明

// let声明（推荐）
let modern = 'new';     // 块级作用域，没有变量提升
// let modern = 'error'; // 不能重复声明

// const声明（常量）
const PI = 3.14159;     // 声明后不能修改
const CONFIG = {        // 对象的属性可以修改
    api: 'http://api.example.com'
};

2. 变量的解构赋值 ​

javascript
// 数组解构
let [a, b, c] = [1, 2, 3];
let [x, ...rest] = [1, 2, 3, 4]; // x=1, rest=[2,3,4]

// 对象解构
let {name, age} = {name: '小明', age: 18};
let {title: bookTitle} = {title: 'JavaScript高级程序设计'};

// 默认值
let [num = 0] = [];     // num = 0
let {sex = '男'} = {};  // sex = '男'

二、作用域 ​

1. 作用域类型 ​

javascript
// 全局作用域
var globalVar = '我是全局变量';
window.globalVar; // 在浏览器环境中可访问

// 函数作用域
function test() {
    var localVar = '我是局部变量';
    console.log(localVar); // 可访问
}
// console.log(localVar); // 报错：无法访问

// 块级作用域
{
    let blockVar = '块级变量';
    const BLOCK_CONST = '块级常量';
}
// console.log(blockVar); // 报错：无法访问

2. 作用域链 ​

javascript
let global = '全局变量';

function outer() {
    let outerVar = '外层变量';
    
    function inner() {
        let innerVar = '内层变量';
        console.log(global);     // 可访问全局变量
        console.log(outerVar);   // 可访问外层变量
        console.log(innerVar);   // 可访问内层变量
    }
    
    inner();
}

outer();

3. 闭包 ​

javascript
function createCounter() {
    let count = 0;  // 私有变量
    
    return {
        increment() {
            return ++count;
        },
        decrement() {
            return --count;
        },
        getCount() {
            return count;
        }
    };
}

const counter = createCounter();
console.log(counter.increment()); // 1
console.log(counter.increment()); // 2
console.log(counter.getCount()); // 2

三、内存管理 ​

1. 内存生命周期 ​

javascript
// 1. 内存分配
let obj = {              // 为对象分配内存
    name: '小明',
    age: 18
};

// 2. 内存使用
console.log(obj.name);   // 使用分配的内存

// 3. 内存释放
obj = null;             // 解除引用，等待垃圾回收

2. 垃圾回收机制 ​

javascript
// 标记清除算法示例
function process() {
    let local = {        // 创建对象，分配内存
        data: 'some data'
    };
    // 函数执行完毕后，local将被标记为不可达
    // 随后的垃圾回收中会被清除
}

// 引用计数算法问题 - 循环引用
function createCycle() {
    let obj1 = {};
    let obj2 = {};
    
    obj1.ref = obj2;    // obj1 引用 obj2
    obj2.ref = obj1;    // obj2 引用 obj1
    
    return 'cycle created';
}
// 即使函数执行完毕，obj1和obj2也不会被回收

3. 内存优化 ​

javascript
// 1. 及时清除引用
function loadData() {
    let heavyData = new Array(10000);
    // 处理数据
    heavyData = null;   // 清除引用
}

// 2. 使用弱引用
const cache = new WeakMap();
let user = {id: 1};
cache.set(user, '用户数据');
user = null;  // user对象可以被垃圾回收

// 3. 避免内存泄漏
function addHandler() {
    const element = document.getElementById('button');
    element.addEventListener('click', () => {
        // 处理点击
    });
    // 记得在适当时候移除事件监听器
}

javascript
// 1. 对象复用
const pool = {
    _pool: [],
    acquire() {
        return this._pool.pop() || {};
    },
    release(obj) {
        this._pool.push(obj);
    }
};

// 2. 分批处理大数据
function processLargeArray(array) {
    const chunk = 1000;
    for(let i = 0; i < array.length; i += chunk) {
        setTimeout(() => {
            // 处理array.slice(i, i + chunk)
        }, 0);
    }
}

// 3. 使用缓存
const memoize = (fn) => {
    const cache = new Map();
    return (...args) => {
        const key = JSON.stringify(args);
        if (cache.has(key)) return cache.get(key);
        const result = fn.apply(this, args);
        cache.set(key, result);
        return result;
    };
};

五、常见问题与解决方案 ​

1. 变量提升问题 ​

javascript
console.log(a);  // undefined
var a = 1;

// 等同于
var a;
console.log(a);
a = 1;

// 使用let/const避免
console.log(b);  // ReferenceError
let b = 2;

2. 闭包内存泄漏 ​

javascript
// 问题代码
function createButtons() {
    let buttons = document.getElementsByTagName('button');
    for(var i = 0; i < buttons.length; i++) {
        buttons[i].onclick = function() {
            console.log(i);  // 总是输出buttons.length
        };
    }
}

// 解决方案
function createButtons() {
    let buttons = document.getElementsByTagName('button');
    for(let i = 0; i < buttons.length; i++) {
        buttons[i].onclick = function() {
            console.log(i);  // 正确输出当前按钮的索引
        };
    }
}

3. this指向问题 ​

javascript
const obj = {
    name: '对象',
    getName() {
        setTimeout(function() {
            console.log(this.name);  // undefined
        }, 100);
    }
};

// 解决方案1：箭头函数
const obj1 = {
    name: '对象',
    getName() {
        setTimeout(() => {
            console.log(this.name);  // '对象'
        }, 100);
    }
};

// 解决方案2：bind
const obj2 = {
    name: '对象',
    getName() {
        setTimeout(function() {
            console.log(this.name);  // '对象'
        }.bind(this), 100);
    }
};

六、ES6+ 作用域与 this 详解 ​

1. 块级作用域的完整应用 ​

javascript
// 1. let 和 const 的块级作用域
{
    let blockScoped = '只在块内可见';
    const BLOCK_CONST = '块级常量';
    
    {
        let nested = '嵌套块级作用域';
        console.log(blockScoped);     // 可访问外层变量
    }
    // console.log(nested);           // ReferenceError
}

// 2. 循环中的块级作用域
for (let i = 0; i < 3; i++) {
    // 每次迭代都会创建新的变量 i
    setTimeout(() => console.log(i), 0);  // 输出 0, 1, 2
}

// 3. switch 语句中的块级作用域
switch (true) {
    case true: {
        let switchVar = 'case 块级作用域';
        break;
    }
    default: {
        let switchVar = '不同的块级作用域';
        break;
    }
}

2. 模块作用域 ​

javascript
// module.js
let moduleVar = '模块私有变量';
export const publicVar = '可导出变量';

// 模块中的顶级 this 是 undefined
console.log(this);  // undefined

// main.js
import { publicVar } from './module.js';
// console.log(moduleVar);  // ReferenceError

3. this 的绑定规则 ​

javascript
// 1. 默认绑定
function showThis() {
    console.log(this);
}
showThis();  // 非严格模式：window，严格模式：undefined

// 2. 隐式绑定
const obj = {
    name: '对象',
    show() {
        console.log(this.name);
    }
};
obj.show();  // '对象'

// 3. 显式绑定
function display() {
    console.log(this.name);
}
const context = { name: '上下文' };
display.call(context);    // '上下文'
display.apply(context);   // '上下文'
const boundFn = display.bind(context);
boundFn();               // '上下文'

// 4. new 绑定
function Person(name) {
    this.name = name;
}
const person = new Person('小明');  // this 绑定到新创建的对象

4. 箭头函数与 this ​

javascript
// 1. 箭头函数继承上下文的 this
const obj = {
    name: '对象',
    traditional: function() {
        setTimeout(function() {
            console.log(this.name);  // undefined
        }, 100);
    },
    arrow: function() {
        setTimeout(() => {
            console.log(this.name);  // '对象'
        }, 100);
    }
};

// 2. 箭头函数不能被改变 this
const arrowFn = () => {
    console.log(this);
};
const context = { name: '上下文' };
arrowFn.call(context);    // this 仍然是定义时的 this

// 3. 箭头函数在对象方法中的陷阱
const bad = {
    name: '对象',
    show: () => {
        console.log(this.name);  // undefined
    }
};

const good = {
    name: '对象',
    show() {
        console.log(this.name);  // '对象'
    }
};

5. class 中的 this ​

javascript
class Example {
    constructor() {
        this.name = '实例';
        // 方法一：构造函数中绑定
        this.show = this.show.bind(this);
    }
    
    // 方法二：类字段语法（推荐）
    display = () => {
        console.log(this.name);
    }
    
    // 普通方法：this 依赖调用方式
    show() {
        console.log(this.name);
    }
    
    // 静态方法中的 this 指向类本身
    static staticMethod() {
        console.log(this === Example);  // true
    }
}

const example = new Example();
const { show, display } = example;

show();     // undefined（解构后丢失 this）
display();  // '实例'（箭头函数保持 this）

6. 实际应用场景 ​

javascript
// 1. React 组件中的 this 绑定
class Component {
    constructor() {
        this.state = { count: 0 };
    }
    
    // 方法一：类字段 + 箭头函数
    handleClick = () => {
        this.setState({ count: this.state.count + 1 });
    }
    
    // 方法二：render 中的箭头函数
    render() {
        return (
            <button onClick={() => this.handleClick()}>
                点击
            </button>
        );
    }
}

// 2. 事件处理中的 this
class EventHandler {
    constructor() {
        this.data = [];
        this.handleEvent = this.handleEvent.bind(this);
    }
    
    handleEvent(event) {
        this.data.push(event);
    }
    
    setup() {
        // 正确：使用已绑定的方法
        document.addEventListener('click', this.handleEvent);
        
        // 正确：使用箭头函数
        document.addEventListener('scroll', (e) => this.handleEvent(e));
    }
}

// 3. 异步操作中的 this
class AsyncExample {
    constructor() {
        this.data = null;
    }
    
    async fetchData() {
        try {
            const response = await fetch('api/data');
            this.data = await response.json();
            
            // 箭头函数保持 this
            setTimeout(() => {
                this.processData();
            }, 1000);
        } catch (error) {
            console.error(error);
        }
    }
    
    processData() {
        console.log(this.data);
    }
}

7. 最佳实践建议 ​

  1. 优先使用箭头函数
javascript
// 在需要保持 this 的场景中使用箭头函数
class Modern {
    data = [];
    handleClick = () => {
        this.data.push('clicked');
    }
}
  1. 避免 this 丢失
javascript
class Prevention {
    constructor() {
        // 在构造函数中进行绑定
        this.handleEvent = this.handleEvent.bind(this);
        
        // 或使用类字段
        this.betterHandle = () => {
            // 这里的 this 永远正确
        };
    }
}
  1. 使用对象方法简写
javascript
// 好的写法
const api = {
    data: [],
    process() {
        // this 指向 api
    }
};

// 避免在对象字面量中使用箭头函数作为方法
const bad = {
    data: [],
    process: () => {
        // this 不指向 bad
    }
};
  1. 注意异步回调
javascript
class Async {
    constructor() {
        this.data = [];
    }
    
    // 好的写法
    async getData() {
        const result = await fetch('/api');
        const data = await result.json();
        
        // 使用箭头函数
        setTimeout(() => {
            this.data = data;
        }, 1000);
    }
}
